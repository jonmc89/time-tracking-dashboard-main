const buttons = {
  daily: document.getElementById("dailyBtn"),
  weekly: document.getElementById("weeklyBtn"),
  monthly: document.getElementById("monthlyBtn"),
};

const elements = {
  current: {
    Work: document.getElementById("workHours"),
    Play: document.getElementById("playHours"),
    Study: document.getElementById("studyHours"),
    Exercise: document.getElementById("exerciseHours"),
    Social: document.getElementById("socialHours"),
    "Self Care": document.getElementById("selfCareHours"),
  },
  previous: {
    Work: document.getElementById("prevWorkHours"),
    Play: document.getElementById("prevPlayHours"),
    Study: document.getElementById("prevStudyHours"),
    Exercise: document.getElementById("prevExerciseHours"),
    Social: document.getElementById("prevSocialHours"),
    "Self Care": document.getElementById("prevSelfCareHours"),
  },
};

function updateUI(timeframe) {
  fetch("./data.json")
    .then((response) => response.json())
    .then((data) => {
      data.forEach((item) => {
        const title = item.title;
        elements.current[title].innerHTML = item.timeframes[timeframe].current;
        elements.previous[title].innerHTML = item.timeframes[timeframe].previous;
      });
    })
    .catch((error) => console.error("Error loading data:", error));

  // Update button styles
  Object.entries(buttons).forEach(([key, btn]) => {
    btn.style.color = key === timeframe ? "white" : "#6f76c8ff";
  });
}

// Event Listeners
buttons.daily.addEventListener("click", () => updateUI("daily"));
buttons.weekly.addEventListener("click", () => updateUI("weekly"));
buttons.monthly.addEventListener("click", () => updateUI("monthly"));

// Initial load
updateUI("daily");
